# 创建项目
npx create-react-app hkzf

# 安装antd-mobile组件库
yarn add antd-mobile-v2

# 安装路由
yarn add react-router-dom
版本`v6`

# 安装axios
yarn add axios

# 安装sass .scss .sass 导入样式文件
yarn add node-sass

# 访问斗鱼接口， 跨域问题
npm install --save-dev http-proxy-middleware
配置setupProxy.js即可
